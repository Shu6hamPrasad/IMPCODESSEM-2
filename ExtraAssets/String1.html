/* Q> Write a program in java to print the following result:
                XUE
                HUA
                PIAO
                PIAO
                BEI
                FENG
                XIAO
                XIAO
            */
            
            public class String1 {
                public static void main(String[] args) {
                    String s = "XUE HUA PIAO PIAO BEI FENG XIAO XIAO";
                    int iterator = 0;
                    for (int i = 0; i < s.length(); i++) {
                        if (s.charAt(i) == 32 || i + 1 == s.length()) {
                            System.out.println(s.substring(iterator, i + 1));
                            iterator = i + 1;
                        }
                    }
                }
            }
