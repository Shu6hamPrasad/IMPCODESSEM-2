public class SelectionSort {
                    public static void main(String[] args) {
                        int[] arr = {12, 13, 11, 15, 9};
                        for (int i = 0; i < arr.length - 1; i++) {
                            for (int j = i + 1; j < arr.length; j++) {
                                if (arr[i] > arr[j]) {
                                    int temp = arr[j];
                                    arr[j] = arr[i];
                                    arr[i] = temp;
                                }
                            }
                        }
                        
                        for (int i : arr) {
                            System.out.println(i);
                        }
                    }
                }
