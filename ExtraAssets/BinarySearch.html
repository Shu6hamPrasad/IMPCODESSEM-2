public class BinarySearch {
                    public static void main(String[] args) {
                        int[] arr = {12, 14, 15, 16};
                        int r = arr.length - 1;
                        int l = 0;
                        int key = 16;
                        int flag = -1;
                        while(l <= r) {
                            int mid = (l + r) / 2;
                            if (arr[mid] == key) {
                                flag = mid;
                                break;
                            }
                            else if (arr[mid] > key)
                                r = mid - 1;
                            else if (arr[mid] < key) 
                                l = mid + 1;
                            else 
                                break;
                        }
                        if (flag != -1)
                            System.out.println("Element found at index: " + flag);
                            
                        else System.out.println("Element not found..."); 
                    }
                }
