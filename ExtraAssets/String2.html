/* Q> Write a program in java to print the following result:
                    X
                    XU
                    XUE
                    XUE_
                    XUE_H
                    XUE_HU
                    XUE_HUA
                     
                   NOTE: '_' INDICATES BLANK SPACE.
                */

                public class String2 {
                    public static void main(String[] args) {
                        String s = "XUE HUA";
                        int k = 0;
                        for (int i = 0; i < s.length(); i++) {
                            System.out.println(s.substring(k, i + 1));
                        }
                    }
                }
