/* Q> Write a program to reverse a String.
                Example:
                Input: XUE HUA PIAO PIAO BEI FENG XIAO XIAO
                Output: OAIX OAIX GNEF IEB OAIP OAIP AUH EUX
            */
            
            public class String6 {
                public static void main(String[] args) {
                    String s = "XUE HUA PIAO PIAO BEI FENG XIAO XIAO";
                    String rev = "";
                    for (int i = s.length() - 1; i >= 0; i--)
                        rev += s.charAt(i);
                    System.out.println(rev);
                }
            }
