public class LinearSearch {
                    public static void main(String[] args) {
                        int[] arr = {0, 1, 2, 3, 5};
                        int x = 5;
                        int flag = -1;
                        for (int i = 0; i < arr.length; i++) {
                            if (arr[i] == x) {
                                flag = i;
                                break;
                            }
                        }
                        
                        if (flag != -1)
                            System.out.println("Element found at index: " + flag);
                            
                        else System.out.println("Element not found..."); 
                    }
                }
