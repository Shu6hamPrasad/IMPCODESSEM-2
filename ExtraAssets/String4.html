/* Q> Write a program in java to print the following pattern:
                XUE HUA PIAO PIAO BEI FENG XIAO XIAO
                XUE HUA PIAO PIAO BEI FENG XIAO
                XUE HUA PIAO PIAO BEI FENG
                XUE HUA PIAO PIAO BEI
                XUE HUA PIAO PIAO
                XUE HUA PIAO
                XUE HUA
                XUE
            */
            
            public class String4 {
                public static void main(String[] args) {
                    String s = "XUE HUA PIAO PIAO BEI FENG XIAO XIAO";
                    int iterator = s.length();
                    for (int i = s.length() - 1; i >= 0; --i) {
                        if (s.charAt(i) == 32 || i - 1 == 0) {
                            System.out.println(s.substring(0, iterator));
                            iterator = i;
                        }
                    }
                }
            }
